api-example.js 
// Initialize Telvergence SDK
const telvergence = require('@telvergence/sdk');

const client = telvergence.init({
  apiKey: process.env.TELVERGENCE_API_KEY,
  encryption: 'military-grade'
});

// Provision a Smart DID
const did = await client.numbers.provision({
  type: 'local',
  areaCode: '512',
  features: ['sms', 'voice', 'e911'],
  smartRouting: true
});

// Set up AI-powered call handling
client.voice.onIncoming(did.number, {
  transcription: true,
  sentiment: true,
  intentRouting: true,
  handler: async (call) => {
    const analysis = await call.analyze();
    if (analysis.sentiment < 0.3) {
      await call.escalate('priority');
    }
  }
});
